    74→cat > "${WRAPPER_DIR}/sudo" << 'SUDOWRAPPER'
    75→#!/bin/bash
    76→# CloudShell sudo wrapper v11 - works in unprivileged containers
    77→# Tries real sudo first, then provides smart alternatives
    78→
    79→# First: try real sudo (works on some platforms)
    80→/usr/bin/sudo "$@" 2>/dev/null && exit 0
    81→
    82→# Second: for apt-get/apt commands, try with real sudo + permission fixes
    83→case "$1" in
    84→    apt|apt-get)
    85→        CMD="$1"; shift
    86→        case "$1" in
    87→            update)
    88→                /usr/bin/sudo apt-get update "$@" 2>/dev/null && exit 0
    89→                echo "⚠ apt-get update not available in this container (unprivileged)"
    90→                echo "  Most packages are pre-installed. For new packages, use:"
    91→                echo "    npm install <pkg>     (Node.js packages)"
    92→                echo "    pip3 install <pkg>    (Python packages)"
    93→                exit 1
    94→                ;;
    95→            install)
    96→                /usr/bin/sudo apt-get install -y "$@" 2>/dev/null && exit 0
    97→                echo "⚠ apt-get install not available in this container (unprivileged)"
    98→                echo "  Package '$*' could not be installed via apt."
    99→                echo "  Try alternatives:"
   100→                echo "    npm install <pkg>     (Node.js packages)"
   101→                echo "    pip3 install <pkg>    (Python packages)"
   102→                exit 1
   103→                ;;
   104→            *)
   105→                /usr/bin/sudo "$CMD" "$@" 2>/dev/null && exit 0
   106→                echo "⚠ apt command not available in this container (unprivileged)"
   107→                exit 1
   108→                ;;
   109→        esac
   110→        ;;
   111→    npm)
   112→        # Handle 'sudo npm install -g' - redirect to user's npm-global
   113→        shift
   114→        if [[ "$1" == "install" ]] || [[ "$1" == "i" ]]; then
   115→            shift
   116→            # Remove -g flag if present and use npm with user prefix
   117→            args=()
   118→            for arg in "$@"; do
   119→                [[ "$arg" != "-g" ]] && [[ "$arg" != "--global" ]] && args+=("$arg")
   120→            done
   121→            npm install -g "${args[@]}"
   122→            exit $?
   123→        fi
   124→        npm "$@"
   125→        exit $?
   126→        ;;
   127→    chown|chmod)
   128→        # Try real sudo for file permissions
   129→        /usr/bin/sudo "$@" 2>/dev/null && exit 0
   130→        # If that fails, try the operation without sudo (may work for user-owned files)
   131→        "$@" 2>/dev/null && exit 0
   132→        echo "⚠ Permission change failed - may need root access"
   133→        exit 1
   134→        ;;
   135→    docker|dockerd*)
   136→        /usr/bin/sudo "$@" 2>/dev/null && exit 0
   137→        "$@" 2>/dev/null && exit 0
   138→        echo "⚠ Docker daemon not running in this container"
   139→        exit 1
   140→        ;;
   141→    systemctl|service)
   142→        echo "⚠ systemctl/service not available in this container (unprivileged)"
   143→        exit 1
   144→        ;;
   145→    *)
   146→        /usr/bin/sudo "$@" 2>/dev/null && exit 0
   147→        exec unshare --user --map-root-user "$@"
   148→        ;;
   149→esac
   150→SUDOWRAPPER
   151→chown cloudshell:cloudshell "${WRAPPER_DIR}/sudo" 2>/dev/null || true

   274→# ─── Auto-refresh PATH before each prompt ────────────────────────
   275→# Most 'curl | bash' installers (opencode, bun, rust, deno, etc.) add a new
   276→# PATH entry to ~/.bashrc but the running shell doesn't pick it up until you
   277→# manually `source ~/.bashrc`. This PROMPT_COMMAND hook checks for any
   278→# newly-created installer directories and adds them to PATH automatically,
   279→# so tools work IMMEDIATELY after install — no `reload` needed.
   280→__cloudshell_refresh_path() {
   281→    local d
   282→    for d in "${HOME}/.opencode/bin" "${HOME}/.bun/bin" "${HOME}/.cargo/bin" \
   283→             "${HOME}/.deno/bin" "${HOME}/.local/bin" "${HOME}/.npm-global/bin" \
   284→             "${HOME}/bin" "${HOME}/go/bin" "${HOME}/.local/go/bin" \
   285→             "${HOME}/.krew/bin" "${HOME}/.yarn/bin"; do
   286→        [ -d "$d" ] || continue
   287→        case ":${PATH}:" in
   288→            *":${d}:"*) ;;
   289→            *) export PATH="${d}:${PATH}" ;;
   290→        esac
   291→    done
   292→    # Also re-source bashrc_env if it changed (for newly-set env vars)
   293→    if [ -f "${HOME}/.bashrc_env" ]; then
   294→        local env_mtime=$(stat -c %Y "${HOME}/.bashrc_env" 2>/dev/null || echo 0)
   295→        if [ -z "${__BASHRC_ENV_MTIME:-}" ] || [ "$env_mtime" -gt "$__BASHRC_ENV_MTIME" ]; then
   296→            source "${HOME}/.bashrc_env" 2>/dev/null
   297→            __BASHRC_ENV_MTIME="$env_mtime"
   298→        fi
   299→    fi
   300→}
   301→# Preserve any existing PROMPT_COMMAND and chain ours
   302→if [ -n "${PROMPT_COMMAND:-}" ]; then
   303→    PROMPT_COMMAND="__cloudshell_refresh_path; ${PROMPT_COMMAND}"
   304→else
   305→    PROMPT_COMMAND="__cloudshell_refresh_path"
   306→fi
   307→export PROMPT_COMMAND

   309→# ─── Reload PATH / source bashrc (after curl|bash installers) ────
   310→# Most 'curl | bash' installers (opencode, bun, rust, deno, etc.)
   311→# add a new PATH entry to ~/.bashrc but the change doesn't take
   312→# effect in the current shell. Run `reload` to fix it instantly.
   313→reload() {
   314→    source "${HOME}/.bashrc" 2>/dev/null
   315→    source "${HOME}/.bashrc_env" 2>/dev/null
   316→    # Also pick up common installer directories that might have been
   317→    # created since the shell started
   318→    for d in "${HOME}/.opencode/bin" "${HOME}/.bun/bin" "${HOME}/.cargo/bin" \
   319→             "${HOME}/.deno/bin" "${HOME}/.local/bin" "${HOME}/.npm-global/bin" \
   320→             "${HOME}/.nvm/versions/node"/*/bin "${HOME}/go/bin" \
   321→             "${HOME}/.local/go/bin" "${HOME}/.krew/bin"; do
   322→        [ -d "$d" ] && case ":$PATH:" in
   323→            *":$d:"*) ;;
   324→            *) export PATH="$d:$PATH" ;;
   325→        esac
   326→    done
   327→    echo "✓ Shell reloaded. PATH updated."
   328→    echo "  PATH entries added: $(echo "$PATH" | tr ':' '\n' | wc -l) directories"
   329→}

   332→# ─── Show what was just installed (where did the binary land?) ──
   333→# Use after running a 'curl | bash' installer to find the new binary
   334→whereis-tool() {
   335→    if [ -z "$1" ]; then
   336→        echo "Usage: whereis-tool <name>"
   337→        echo "  Searches common install locations for the given tool"
   338→        echo "  Example: whereis-tool opencode"
   339→        return 1
   340→    fi
   341→    local name="$1"
   342→    local found=0
   343→    echo "Searching for '$name' in common install locations..."
   344→    for d in "${HOME}/.local/bin" "${HOME}/.npm-global/bin" "${HOME}/bin" \
   345→             "${HOME}/.opencode/bin" "${HOME}/.bun/bin" "${HOME}/.cargo/bin" \
   346→             "${HOME}/.deno/bin" "${HOME}/go/bin" "${HOME}/.local/go/bin" \
   347→             "/usr/local/bin" "/usr/bin"; do
   348→        if [ -x "${d}/${name}" ]; then
   349→            echo "  ✓ ${d}/${name}"
   350→            found=1
   351→        fi
   352→    done
   353→    if [ $found -eq 0 ]; then
   354→        echo "  ✗ '$name' not found in any common location."
   355→        echo "    Try: reload  (to refresh PATH after an installer ran)"
   356→        echo "    Or:  find ~ -name '$name' -type f 2>/dev/null"
   357→    else
   358→        echo ""
   359→        echo "  To use it: reload  (or open a new terminal tab)"
   360→    fi
   361→}

   447→# ─── Internal: update a single env var in ~/.bashrc_env ─────
   448→_claude_update_env() {
   449→    local VAR_NAME="$1"
   450→    local VALUE="$2"
   451→    local ENV_FILE="${HOME}/.bashrc_env"
   452→
   453→    # Remove old line for this variable (if exists)
   454→    if [ -f "$ENV_FILE" ]; then
   455→        sed -i "/^export ${VAR_NAME}=/d" "$ENV_FILE" 2>/dev/null || true
   456→    fi
   457→
   458→    # Append new value
   459→    echo "export ${VAR_NAME}=\"${VALUE}\"" >> "$ENV_FILE"
   460→}
